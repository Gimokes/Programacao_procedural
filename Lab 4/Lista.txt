#include <stdio.h>

void main()
{
    int a=100;
    int max=0;
    while (a<1000)
    {
        int b=100;
        while(b<1000)
        {
            int produto=a*b;
            int reverso=0;
            int temp=produto;
            while(temp>0)
            {
                reverso=reverso*10+temp%10;
                temp/=10;
            }
            if(produto==reverso&&produto>max)
            {
                max=produto;
            }
            b++;
        }
        a++;
    }
    printf("%d\n",max);
}
