#include <stdio.h>
int contadorprimos(int n);

void main()
{
    int n;
    scanf("%d",&n);
    printf("%d\n",contadorprimos(n));
}
int contadorprimos(int n)
{
    int a,b, contador=0;
    for(a=2;a<n;a++)
    {
        for(b=2;b<a;b++)
        {
            if(a%b==0)
            {
                break;
            }
        }
        if(b==a)
        {
            contador++;
        }
    }
    return contador;
}

#include <stdio.h>
void triangulo(int n);
void main()
{
    int n;
    scanf("%d", &n);
    triangulo(n);
}
void triangulo(int n)
{
    int a,b;
    for(a=1;a<=2*n-1;a++)
    {
        if(a<=n)
        {
            for(b=1;b<=a;b++)
            printf("*");
        }
        else
        {
            for(b=1;b<=2*n-a;b++)
            printf("*");
        }
        printf("\n");
    }
}

void triangulo(int n)
{
    int a,b;
    for(a=1;a<=n;a++)
    {
        for(b=1;b<=n-a;b++)
        printf(" ");
        for(b=1;b<=2*a-1;b++)
        printf("*");
        printf("\n");
    }
}
