#include <stdio.h>
#include <stdlib.h>

int main()
{
  int horas;
  int minutos;
  int segundos;
  int tempoEmSegundos;
  int duracao;
  int tempoFinal; 
  int sobra;
  printf("Digite o horario de inicio: \n");
  scanf("%d:%d:%d", &horas, &minutos, &segundos);
  printf("Digite a duracao em segundos: \n");
  scanf("%d", &duracao);

  tempoEmSegundos = horarioParaSegundos(horas, minutos, segundos);
  tempoFinal = tempoEmSegundos + duracao;

  horas = (tempoFinal - (tempoFinal % 3600)) / 3600;
  sobra = tempoFinal % 3600;
  minutos = (sobra - (sobra % 60)) / 60;
  sobra = sobra % 60;
  tempoFinal = sobra;

  printf("O horario de termino eh as %d hora(s), %d minuto(s) e %d segundo(s)", horas, minutos, tempoFinal);

  return 0;
}

int horarioParaSegundos(int h, int m, int s)
{
  return h * 3600 + m * 60 + s;
}
