#include <stdio.h>
int main(void)
{
    int A[6];
    for(int x=0;x<6;x++)
    {
        printf("insira o %d valor:",x+1);
        scanf("%d",&A[x]);
    }
    int soma_par=0;
    int qtd_impar=0;
    for(int x=0;x<6;x++)
    {
        if(A[x]%2==0)
        {
            printf("par:%d\n",A[x]);
            soma_par+=A[x];
        }
        else
        {
            printf("impar: %d\n", A[x]);
            qtd_impar++;
        }
    }
    printf("soma_par:%d\n",soma_par);
    printf("qtd_impar:%d\n",qtd_impar);
    return 0;
}

#include <stdio.h>
int main (void)
{
    int a[10];
    for(int x=0;x<10;x++)
    {
        int valor;
        printf("insira o valor %d valor: ",x+1);
        scanf("%d", &valor);
        int existe=0;
        for(int y=0;y<x;y++)
        {
            if(a[y]==valor)
            {
                existe=1;
                break;
            }
        }
        if(existe)
        {
            x--;
            continue;
        }
        a[x]=valor;
    }
    for(int x=0;x<10;x++)
    {
        printf("%d\n",a[x]);
    }
    return 0;
}
